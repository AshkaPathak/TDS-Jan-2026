# GA4 — Q17: LLM Hallucination Trap Matrix

## Problem Summary

We are given a ZIP archive containing **1000 Python scripts** generated by a misaligned AI model.

- **999 scripts contain hallucinated method calls**
- **1 script is completely valid**

A hallucination refers to a **plausible but non-existent method or function**.

Examples include:

- `df.drop_nulls()` instead of `df.dropna()`
- `json.parse()` instead of `json.loads()`
- `os.path.build_path()` instead of `os.path.join()`

Our objective is to **automatically identify the single valid script** that contains **no hallucinated method calls**.

Manually inspecting 1000 scripts would be inefficient, so we solve the problem using **automated static code analysis**.

---

# Key Insight

Hallucinated scripts typically introduce one or more of the following:

- Non-existent module attributes
- Undefined functions
- Incorrect imports
- Invalid argument usage

Using Python’s **AST (Abstract Syntax Tree)** we can parse each file **without executing it** and detect these errors automatically.

The correct script will therefore be **the only script with zero detected issues**.

# Step 3 — Static Analysis Implementation

Paste the following Python code.

```python
import ast
import os
import builtins
import importlib
import datetime

TIMEDELTA_KW = {"days","seconds","microseconds","milliseconds","minutes","hours","weeks"}

BUILTINS = set(dir(builtins))

def load_module(name):
    try:
        return importlib.import_module(name)
    except:
        return None

def resolve_chain(expr):
    attrs=[]
    while isinstance(expr,ast.Attribute):
        attrs.append(expr.attr)
        expr=expr.value
    if isinstance(expr,ast.Name):
        return expr.id,list(reversed(attrs))
    return None,None

def traverse_attr(obj,attrs):
    cur=obj
    for a in attrs:
        if cur is None or not hasattr(cur,a):
            return None
        cur=getattr(cur,a)
    return cur

def build_import_env(tree):
    env={}
    for node in ast.walk(tree):
        if isinstance(node,ast.Import):
            for alias in node.names:
                name=alias.asname or alias.name.split(".")[-1]
                env[name]=("module",alias.name)
        elif isinstance(node,ast.ImportFrom):
            if not node.module:
                continue
            mod=load_module(node.module)
            for alias in node.names:
                name=alias.asname or alias.name
                obj=getattr(mod,alias.name) if mod else None
                env[name]=("object",obj)
    return env

def get_defined_names(tree,import_env):
    defined=set(import_env.keys())
    for node in ast.walk(tree):
        if isinstance(node,(ast.FunctionDef,ast.AsyncFunctionDef,ast.ClassDef)):
            defined.add(node.name)
        elif isinstance(node,ast.arg):
            defined.add(node.arg)
        elif isinstance(node,ast.Assign):
            for t in node.targets:
                if isinstance(t,ast.Name):
                    defined.add(t.id)
        elif isinstance(node,ast.AnnAssign):
            if isinstance(node.target,ast.Name):
                defined.add(node.target.id)
    return defined

def analyze_file(path):
    src=open(path).read()

    try:
        tree=ast.parse(src)
    except SyntaxError:
        return ["syntax_error"]

    import_env=build_import_env(tree)
    defined=get_defined_names(tree,import_env)

    errors=[]

    def check_attr(expr):
        root,attrs=resolve_chain(expr)
        if not root or not attrs:
            return

        if root in import_env and import_env[root][0]=="module":
            m=load_module(import_env[root][1])
            if traverse_attr(m,attrs) is None:
                errors.append("missing_module_attr")

        if root in import_env and import_env[root][0]=="object":
            base=import_env[root][1]
            if traverse_attr(base,attrs) is None:
                errors.append("missing_imported_attr")

    class Visitor(ast.NodeVisitor):

        def visit_Call(self,node):

            if isinstance(node.func,ast.Name):
                fn=node.func.id
                if fn not in defined and fn not in BUILTINS:
                    errors.append("undefined_callable")

            if isinstance(node.func,ast.Attribute):
                check_attr(node.func)

            self.generic_visit(node)

    Visitor().visit(tree)
    return errors

def main():

    good=[]

    for fname in sorted(os.listdir("scripts")):

        if not fname.endswith(".py"):
            continue

        path=os.path.join("scripts",fname)

        errs=analyze_file(path)

        if not errs:
            good.append(fname)

    print("Valid candidates:",good)
    print("Count:",len(good))

if __name__=="__main__":
    main()
```
